F)Create a function to find the factorial of a number.

create or replace function f1(n number)
return number
as
f number:=1;
i number:=1;
begin
while i<=n loop
f:=f*i;
i:=i+1;
end loop;
return f;
end;

select f1(5) from dual;

I)Create a function to display fibonacci series upto first 10 numbers.

create or replace function f1
return varchar
as
n1 number:=0;
n2 number:=1;
n3 number;
result varchar(4000):=n1||','||n2;
begin
for i in 3..10 loop
n3:=n1+n2;
result:=result||','||n3;
n1:=n2;
n2:=n3;
end loop;
return result;
end;

select f1 from dual;
