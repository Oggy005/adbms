F)Create a function to find the factorial of a number.

create or replace function f1(n number)
return number
as
f number:=1;
begin
if n<0 then
return null;
end if;
for i in 1..n loop
f:=f*i;
end loop;
return f;
end;

select f1(3) from dual;
